Aperçu du Code


const { Client, GatewayIntentBits } = require('discord.js');
const { token } = require('./config.json');
const client = new Client({
    intents: [
        GatewayIntentBits.Guilds,
        GatewayIntentBits.GuildMessages
    ]
});

client.once('ready', () => {
    console.log('Bot is ready!');
});

client.on('interactionCreate', async interaction => {
    if (!interaction.isCommand()) return;

    const { commandName } = interaction;

    if (commandName === 'random-identity') {
        const champion = getRandomChampion();
        const role = getRandomRole();
        
        await interaction.reply({
            content: `Votre identité: ${champion} - ${role}`,
            ephemeral: true
        });
    }
});

client.login(token);
                        

const { SlashCommandBuilder } = require('@discordjs/builders');
const { getRandomChampion, getRandomRole, getChampionInfo } = require('./utils');

module.exports = {
    randomIdentity: new SlashCommandBuilder()
        .setName('random-identity')
        .setDescription('Génère une identité LoL aléatoire')
        .toJSON(),

    championInfo: new SlashCommandBuilder()
        .setName('champion-info')
        .setDescription('Affiche les informations d\'un champion spécifique')
        .addStringOption(option =>
            option.setName('champion')
                .setDescription('Le nom du champion')
                .setRequired(true))
        .toJSON(),

    help: new SlashCommandBuilder()
        .setName('help')
        .setDescription('Liste toutes les commandes disponibles')
        .toJSON(),

    async executeRandomIdentity(interaction) {
        const champion = await getRandomChampion();
        const role = getRandomRole();
        await interaction.reply(`Votre identité LoL : ${champion} - ${role}`);
    },

    async executeChampionInfo(interaction) {
        const championName = interaction.options.getString('champion');
        const championInfo = await getChampionInfo(championName);
        if (championInfo) {
            await interaction.reply({ embeds: [championInfo] });
        } else {
            await interaction.reply('Champion non trouvé. Vérifiez l\'orthographe et réessayez.');
        }
    },

    async executeHelp(interaction) {
        const commands = [
            '`/random-identity` : Génère une identité LoL aléatoire',
            '`/champion-info [nom du champion]` : Affiche les informations d\'un champion spécifique',
            '`/help` : Affiche cette liste de commandes'
        ];
        await interaction.reply(`Commandes disponibles :\n${commands.join('\n')}`);
    }
};
                        

async function getChampionData() {
    const response = await fetch(
        'http://ddragon.leagueoflegends.com/cdn/13.1.1/data/fr_FR/champion.json'
    );
    const data = await response.json();
    return Object.values(data.data);
}

async function getRandomChampion() {
    const champions = await getChampionData();
    const randomIndex = Math.floor(Math.random() * champions.length);
    return champions[randomIndex].name;
}

function getRandomRole() {
    const roles = ['Top', 'Jungle', 'Mid', 'ADC', 'Support'];
    return roles[Math.floor(Math.random() * roles.length)];
}
