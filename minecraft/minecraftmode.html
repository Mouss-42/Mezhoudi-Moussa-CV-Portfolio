public class BattleRoyalePlugin extends JavaPlugin {
    @Override
    public void onEnable() {
        getLogger().info("Battle Royale plugin has been enabled!");
        getCommand("start").setExecutor(new StartCommand(this));
        getServer().getPluginManager().registerEvents(new GameListener(this), this);
    }

    @Override
    public void onDisable() {
        getLogger().info("Battle Royale plugin has been disabled!");
    }
}
                        

public class Game {
    private final Set players;
    private final BukkitRunnable gameLoop;

    public Game(Set players) {
        this.players = players;
        this.gameLoop = new BukkitRunnable() {
            @Override
            public void run() {
                if (players.size() <= 1) {
                    endGame();
                    cancel();
                } else {
                    giveRandomItems();
                }
            }
        };
    }

    public void start() {
        teleportPlayers();
        gameLoop.runTaskTimer(BattleRoyalePlugin.getPlugin(BattleRoyalePlugin.class), 0L, 200L);
    }

    private void giveRandomItems() {
        for (Player player : players) {
            ItemManager.giveRandomItem(player);
        }
    }

    private void endGame() {
        // Announce winner and clean up
    }
}
                        

public class ItemManager {
    private static final List ITEMS = Arrays.asList(
        new ItemStack(Material.DIAMOND_SWORD),
        new ItemStack(Material.BOW),
        new ItemStack(Material.ARROW, 64),
        new ItemStack(Material.GOLDEN_APPLE, 3)
    );

    public static void giveRandomItem(Player player) {
        ItemStack item = ITEMS.get(new Random().nextInt(ITEMS.size()));
        player.getInventory().addItem(item);
        player.sendMessage("You received a " + item.getType().name() + "!");
    }
}
